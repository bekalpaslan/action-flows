// Backend JSONL tail service const tail = spawn('tail', ['-f', sessionJSONLPath]); tail.stdout.on('data', (data) => { const line = data.toString(); try { const json = JSON.parse(line); if (json.type === 'assistant' && json.content) { const gateMatch = json.content.match(/\[GATE:(\d+)\]\s*(\{.*?\})/); if (gateMatch) { const gateLog = JSON.parse(gateMatch[2]); await gateLogger.store(gateLog); broadcastService.emit('gate:logged', gateLog); } } } catch (err) { console.error('Parse error:', err); } });
