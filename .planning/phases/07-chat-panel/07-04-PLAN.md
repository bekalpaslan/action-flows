         useEffect(() => {

               const messages = (envelope.payload as any)?.messages ?? [];
               // Normalize workbenchId comparison: the WS payload uses string,
               // ensure match even if casing/format differs (Issue #4: workbenchId typing)
               setHistoryEntries(messages.slice(0, 20).map((m: any) => ({

                 content: typeof m.content === 'string' ? m.content : JSON.stringify(m.content ?? ''),
                 timestamp: m.timestamp ?? m.created_at ?? undefined,
               })));
               setHistoryLoading(false);
             }
           });
           return unsub;
         }, [workbenchId]);
