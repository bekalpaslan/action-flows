The `handleAutonomyChange` function:
```typescript
const handleAutonomyChange = async (workbenchId: WorkbenchId, level: AutonomyLevel) => {
  // Optimistic UI update
  useValidationStore.getState().setAutonomyLevel(workbenchId, level);

  try {
    const res = await fetch(`/api/approvals/autonomy/${workbenchId}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ level }),
    });
    if (res.ok) {
      toast.success(`${WORKBENCHES.find(w => w.id === workbenchId)?.label} autonomy set to ${level.charAt(0).toUpperCase() + level.slice(1)}`);
    } else {
      toast.error('Failed to update autonomy level');
    }
  } catch {
    toast.error('Failed to update autonomy level');
  }
};
```
